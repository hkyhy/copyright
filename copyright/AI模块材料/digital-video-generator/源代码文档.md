const generateAccessToken = async (fastify, forceRefresh = false) => {
  // 检查缓存是否有效（提前5分钟刷新）
  const now = Date.now();
  if (!forceRefresh && tokenCache.token && tokenCache.expiresAt > now + 5 * 60 * 1000) {
    console.log('使用缓存的 access_token');
    return tokenCache.token;
  }
  // 从环境变量获取配置
  const appId = fastify.config.CHANJING_APPID;
  const secretKey = fastify.config.CHANJING_SEC;
  const baseUrl = fastify.config.CHANJING_BASEURL || 'https://open-api.chanjing.cc';
  if (!appId || !secretKey) {
    throw new Error('蝉镜API配置不完整，请检查 CHANJING_APPID 和 CHANJING_SEC 环境变量');
  }
  try {
    console.log('正在获取新的 access_token...');
    // 调用蝉镜API获取 access_token
    const response = await axios.post(
      `${baseUrl}/open/v1/access_token`,
      {
        app_id: appId,
        secret_key: secretKey
      },
      {
        headers: {
          'Content-Type': 'application/json'
        }
      }
    );
