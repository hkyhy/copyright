const Authenticate = createWithRemoteLoader({
  modules: ['components-core:Global@SetGlobal', 'components-core:Global@usePreset']

  const { apis } = usePreset();

    <Fetch
      {...Object.assign({}, apis.interview.getUserInfo, {
        options: {

        return (
          <SetGlobal globalKey="userInfo" value={{ value: data, reload }} needReady>

export const createGetVolume = stream => {
  if (!stream) {
    return () => 0;
  }
  const audioContext = new (window.AudioContext || window.webkitAudioContext)();
  const analyser = audioContext.createAnalyser();
  const microphone = audioContext.createMediaStreamSource(stream);
  microphone.connect(analyser);
  analyser.fftSize = 2048; // 设置 FFT 大小
  const bufferLength = analyser.frequencyBinCount;
  const dataArray = new Uint8Array(bufferLength);
  return () => {
    analyser.getByteTimeDomainData(dataArray); // 获取时域数据
    let sum = 0;
    // 计算音量
    for (let i = 0; i < bufferLength; i++) {
      sum += (dataArray[i] - 128) * (dataArray[i] - 128); // 计算平方和
    }
    const rms = Math.sqrt(sum / bufferLength); // 计算 RMS（均方根）
    const volume = Math.min(Math.max((rms / 128) * 100, 0) * 2, 100);
    return volume;
  };
